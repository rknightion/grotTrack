PHASE 2: WINDOW/APP TRACKING WITH ACCESSIBILITY API + CGWindowList [COMPLETE]
================================================================================
Goal: Track which app and window is in the foreground. Use CGWindowListCopyWindowInfo
to enumerate visible windows. Display in the menu bar popover in real-time.

Step 2.1: Create PermissionManager
  - Create file: Services/PermissionManager.swift
  - @Observable @MainActor class with properties:
    - accessibilityGranted: Bool = false
    - screenRecordingGranted: Bool = false
  - Method checkAccessibility() -> Bool:
    - import ApplicationServices
    - Call AXIsProcessTrusted()
    - Update accessibilityGranted
    - Return result
  - Method requestAccessibility():
    - let options = [kAXTrustedCheckOptionPrompt.takeRetainedValue(): true] as CFDictionary
    - AXIsProcessTrustedWithOptions(options)
    - This shows the system dialog directing user to System Settings
  - Method checkAllPermissions():
    - Call checkAccessibility()
    - (Screen recording check added in Phase 4)

Step 2.2: Create VisibleWindowTracker
  - Create file: Services/VisibleWindowTracker.swift
  - Uses CGWindowListCopyWindowInfo to enumerate visible windows (no JXA needed):
    ```swift
    import CoreGraphics

    struct VisibleWindow: Sendable {
        let ownerName: String
        let ownerPID: pid_t
        let title: String
        let bounds: CGRect
        let layer: Int
    }

    @MainActor
    final class VisibleWindowTracker {
        func visibleWindows() -> [VisibleWindow] {
            let options: CGWindowListOption = [.optionOnScreenOnly, .excludeDesktopElements]
            guard let infoList = CGWindowListCopyWindowInfo(options, kCGNullWindowID)
                    as? [[String: Any]] else { return [] }
            return infoList.compactMap { info in
                guard let name = info[kCGWindowOwnerName as String] as? String,
                      let pid = info[kCGWindowOwnerPID as String] as? pid_t,
                      let title = info[kCGWindowName as String] as? String,
                      let boundsDict = info[kCGWindowBounds as String] as? [String: CGFloat],
                      let layer = info[kCGWindowLayer as String] as? Int,
                      layer == 0  // normal windows only
                else { return nil }
                let bounds = CGRect(x: boundsDict["X"] ?? 0,
                                    y: boundsDict["Y"] ?? 0,
                                    width: boundsDict["Width"] ?? 0,
                                    height: boundsDict["Height"] ?? 0)
                return VisibleWindow(ownerName: name, ownerPID: pid,
                                     title: title, bounds: bounds, layer: layer)
            }
        }

        /// Count of distinct apps with visible windows right now.
        func visibleAppCount() -> Int {
            Set(visibleWindows().map(\.ownerName)).count
        }
    }
    ```

Step 2.3: Create ActivityTracker
  - Create file: Services/ActivityTracker.swift
  - @Observable @MainActor class with properties:

    - Update appState: currentAppName, currentWindowTitle

PHASE 3: CHROME EXTENSION FOR BROWSER TABS [COMPLETE]
================================================================================
Goal: Get active browser tab title + URL via a Chrome extension communicating
with a native messaging host, replacing the old JXA/BrowserTabReader approach.
This avoids the Automation permission entirely.

Utilise WXT (https://wxt.dev/) as the extension SDK to support common functionality. 

Step 3.1: Create Chrome Extension (MV3)
  - Create directory: GrotTrackExtension/
  - Create file: GrotTrackExtension/manifest.json
    ```json
    {
      "manifest_version": 3,
      "name": "GrotTrack Tab Tracker",
      "version": "1.0",
      "description": "Sends active tab info to GrotTrack native app",
      "permissions": ["tabs", "nativeMessaging"],
      "background": {
        "service_worker": "background.js"
      }
    }
    ```

Step 3.2: Create background.js
  - Create file: GrotTrackExtension/background.js
  - Monitors active tab changes and sends title+URL to native host:
    ```javascript
    const HOST_NAME = "com.grottrack.tabtracker";

    function sendTab(tab) {
      if (!tab || !tab.url) return;
      chrome.runtime.sendNativeMessage(
        HOST_NAME,
        { title: tab.title || "", url: tab.url },
        (response) => {
          if (chrome.runtime.lastError) {
            console.warn("Native messaging error:", chrome.runtime.lastError.message);
          }
        }
      );
    }

    chrome.tabs.onActivated.addListener((activeInfo) => {
      chrome.tabs.get(activeInfo.tabId, (tab) => {
        if (tab) sendTab(tab);
      });
    });

    chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
      if (changeInfo.status === "complete" && tab.active) {
        sendTab(tab);
      }
    });

    chrome.windows.onFocusChanged.addListener((windowId) => {
      if (windowId === chrome.windows.WINDOW_ID_NONE) return;
      chrome.tabs.query({ active: true, windowId: windowId }, (tabs) => {
        if (tabs.length > 0) sendTab(tabs[0]);
      });
    });
    ```

Step 3.3: Create NativeMessageHost
  - Create file: Services/NativeMessageHost.swift
  - Reads Chrome native messaging protocol: 32-bit little-endian length prefix + JSON:
    ```swift
    import Foundation

    actor NativeMessageHost {
        private var inputPipe: FileHandle?
        private var onMessage: (@Sendable (BrowserTabMessage) -> Void)?

        struct BrowserTabMessage: Codable, Sendable {
            let title: String
            let url: String
        }

        func start(onMessage: @escaping @Sendable (BrowserTabMessage) -> Void) {
            self.onMessage = onMessage
            Task {
                await listenForMessages()
            }
        }

        private func listenForMessages() {
            let stdin = FileHandle.standardInput
            while true {
                // Read 4-byte length prefix (little-endian UInt32)
                let lengthData = stdin.readData(ofLength: 4)
                guard lengthData.count == 4 else { break }
                let length = lengthData.withUnsafeBytes {
                    $0.load(as: UInt32.self)
                }
                guard length > 0, length < 1_000_000 else { continue }

                // Read JSON payload
                let jsonData = stdin.readData(ofLength: Int(length))
                guard jsonData.count == Int(length),
                      let message = try? JSONDecoder().decode(
                          BrowserTabMessage.self, from: jsonData)
                else { continue }

                onMessage?(message)
            }
        }

        /// Send a response back to Chrome (length-prefixed JSON).
        func sendResponse(_ dict: [String: String]) throws {
            let data = try JSONEncoder().encode(dict)
            var length = UInt32(data.count)
            let lengthData = Data(bytes: &length, count: 4)
            FileHandle.standardOutput.write(lengthData)
            FileHandle.standardOutput.write(data)
        }
    }
    ```

Step 3.4: Create BrowserTabService
  - Create file: Services/BrowserTabService.swift
  - @Observable @MainActor class that stores the latest tab info:
    ```swift
    import Foundation

    @Observable
    @MainActor
    final class BrowserTabService {
        var latestTitle: String = ""
        var latestURL: String = ""
        var lastUpdated: Date?

        /// Called from NativeMessageHost when Chrome sends a tab update.
        func update(title: String, url: String) {
            latestTitle = title
            latestURL = url
            lastUpdated = Date()
        }

        /// Returns current tab info if it was updated within the last 10 seconds.
        func currentTab() -> (title: String, url: String)? {
            guard let updated = lastUpdated,
                  Date().timeIntervalSince(updated) < 10 else { return nil }
            return (latestTitle, latestURL)
        }
    }
    ```

Step 3.5: Create native messaging host manifest
  - Create file: NativeMessagingHost/com.grottrack.tabtracker.json
  - This file is installed to
    ~/Library/Application Support/Google/Chrome/NativeMessagingHosts/
    ```json
    {
      "name": "com.grottrack.tabtracker",
      "description": "GrotTrack native messaging host for browser tab tracking",
      "path": "/Applications/GrotTrack.app/Contents/MacOS/GrotTrackTabHost",
      "type": "stdio",
      "allowed_origins": [
        "chrome-extension://<EXTENSION_ID>/"
      ]
    }
    ```
  - Note: <EXTENSION_ID> must be replaced with the actual extension ID after
    loading it in Chrome. During development, load unpacked from GrotTrackExtension/
    and copy the ID from chrome://extensions.

Step 3.6: Wire BrowserTabService into ActivityTracker
  - In ActivityTracker.pollCurrentWindow():
    - If frontmost app bundleID is "com.google.Chrome" or
      "com.google.Chrome.canary":
      - Call browserTabService.currentTab()
      - If non-nil, update appState.currentBrowserTab with title
    - Otherwise clear appState.currentBrowserTab

Step 3.7: Create install helper
  - Create file: Utilities/ChromeExtensionInstaller.swift
  - Method installNativeHost():
    - Copy com.grottrack.tabtracker.json to
      ~/Library/Application Support/Google/Chrome/NativeMessagingHosts/
    - Substitute the actual path to the native host binary
  - Method checkInstallation() -> Bool:
    - Verify the manifest file exists at the expected location

VERIFY: Build and run, then load GrotTrackExtension/ as unpacked extension in Chrome.
  - Open Chrome, navigate to a website
  - Menu bar popover should show Chrome tab title + URL
  - Switch tabs in Chrome -> title should update within seconds
  - Switch to a non-browser app -> browser tab info should clear
  - Verify no Automation permission is requested at any point

    - private var captureTimer: Timer?

    - var maxWidth: CGFloat = 1280.0
    - var jpegQuality: CGFloat = 0.8
    - var thumbnailWidth: CGFloat = 320.0
    - var screenshotsDir: URL (computed from app support directory)
    - var thumbnailsDir: URL (computed from app support directory)
  - init():
    - Create directories if they don't exist:
      ~/Library/Application Support/GrotTrack/Screenshots/
      ~/Library/Application Support/GrotTrack/Thumbnails/
  - Method startCapturing():
    - Timer.scheduledTimer repeating every screenshotInterval
    - On each tick, call captureScreenshot()
  - Method stopCapturing():
    - Invalidate timer
  - Method captureScreenshot() async:
    - import ScreenCaptureKit
    - Get available content: let content = try await SCShareableContent.current
    - Get primary display: content.displays.first
    - Create filter: SCContentFilter(display: display, excludingWindows: [])
    - Create config: SCStreamConfiguration() with desired resolution
    - Capture: let image = try await SCScreenshotManager.captureImage(
        contentFilter: filter, configuration: config)
    - Resize to maxWidth
    - Generate date-based path: Screenshots/YYYY-MM-DD/HH-MM-SS.jpg
    - Save JPEG data to path
    - Generate thumbnail (320px) and save to Thumbnails/
    - Return (screenshotPath, thumbnailPath, fileSize)
  - Method ensureDirectories(for date: Date):
    - Create Screenshots/YYYY-MM-DD/ and Thumbnails/YYYY-MM-DD/ if missing

Step 4.3: Wire screenshot capture to tracking
  - In GrotTrackApp.swift, create ScreenshotManager instance
  - When activityTracker.startTracking() is called, also start screenshot capture
  - When tracking stops, stop screenshot capture
  - In MenuBarView, optionally show "Last screenshot: X seconds ago"

VERIFY: Build and run, grant Screen Recording permission.
  - After 30 seconds, check ~/Library/Application Support/GrotTrack/Screenshots/
  - JPEG files should appear in YYYY-MM-DD folders
  - Each file should be ~50-100KB
  - Thumbnails should appear in Thumbnails/ folder at ~5-15KB each
  - Files should be viewable and show screen content

PHASE 7: MULTITASKING DETECTION [COMPLETE]
================================================================================
Goal: Detect rapid window switching and score each time block. Use CGWindowList
visible-window count to enrich scoring.

Step 7.1: Create MultitaskingDetector
  - Create file: Services/MultitaskingDetector.swift
  - @Observable @MainActor class:
    - var currentScore: Double = 0.0 (0.0=focused, 1.0=heavy multitasking)
    - private var switchHistory: [(timestamp: Date, bundleID: String)] = []
    - private let rollingWindowSeconds: TimeInterval = 300 (5 minutes)
    - private let visibleWindowTracker: VisibleWindowTracker
  - init(visibleWindowTracker: VisibleWindowTracker):
    - Store reference to the shared VisibleWindowTracker from Phase 2
  - Method recordSwitch(bundleID: String, at timestamp: Date = Date()):
    - Append to switchHistory
    - Prune entries older than rollingWindowSeconds
    - currentScore = calculateScore()
  - Method calculateScore() -> Double:
    - let uniqueApps = Set(switchHistory.map(\.bundleID)).count
    - let switchCount = switchHistory.count
    - let windowMinutes = rollingWindowSeconds / 60.0
    - let switchRate = Double(switchCount) / windowMinutes
    - let visibleApps = visibleWindowTracker.visibleAppCount()
    - Enrich with visible-window count:
      ```swift
      let switchComponent = (switchRate / 20.0) * 0.4
      let uniqueComponent = (Double(uniqueApps) / 10.0) * 0.3
      let visibleComponent = (Double(min(visibleApps, 8)) / 8.0) * 0.3
      let rawScore = switchComponent + uniqueComponent + visibleComponent
      return min(max(rawScore, 0.0), 1.0)
      ```

  - Simple wrapper around Security framework:
    ```swift
    import Security

    enum Keychain {
        static func save(key: String, value: String) throws {
            let data = value.data(using: .utf8)!
            let query: [String: Any] = [
                kSecClass as String: kSecClassGenericPassword,
                kSecAttrAccount as String: key,
                kSecAttrService as String: "com.grottrack.api",
                kSecValueData as String: data
            ]
            SecItemDelete(query as CFDictionary) // Remove old value
            let status = SecItemAdd(query as CFDictionary, nil)
            guard status == errSecSuccess else {
                throw KeychainError.saveFailed(status)
            }
        }

        static func load(key: String) -> String? {
            let query: [String: Any] = [
                kSecClass as String: kSecClassGenericPassword,
                kSecAttrAccount as String: key,
                kSecAttrService as String: "com.grottrack.api",
                kSecReturnData as String: true,
                kSecMatchLimit as String: kSecMatchLimitOne
            ]
            var result: AnyObject?
            let status = SecItemCopyMatching(query as CFDictionary, &result)
            guard status == errSecSuccess, let data = result as? Data else { return nil }
            return String(data: data, encoding: .utf8)
        }

        static func delete(key: String) {
            let query: [String: Any] = [
                kSecClass as String: kSecClassGenericPassword,
                kSecAttrAccount as String: key,
                kSecAttrService as String: "com.grottrack.api"
            ]
            SecItemDelete(query as CFDictionary)
        }
    }

    enum KeychainError: Error {
        case saveFailed(OSStatus)
    }
    ```
